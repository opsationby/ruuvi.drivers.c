54 rd_status_t ri_radio_init (const ri_radio_modulation_t modulation)
55 {
56  rd_status_t status = RD_SUCCESS;
57  ret_code_t err_code = NRF_SUCCESS;
58 
59  if (ri_radio_is_init())
60  {
61  status |= RD_ERROR_INVALID_STATE;
62  }
63  else
64  {
65  err_code = nrf_sdh_enable_request();
66  RD_ERROR_CHECK (err_code, NRF_SUCCESS);
67  // Configure the BLE stack using the default settings.
68  // Fetch the start address of the application RAM.
69  uint32_t ram_start = 0;
70  // As of SD 6.1.1, only one advertising configuration is allowed.
71  err_code |= nrf_sdh_ble_default_cfg_set (RUUVI_NRF5_SDK15_BLE4_STACK_CONN_TAG,
72  &ram_start);
73  RD_ERROR_CHECK (err_code, NRF_SUCCESS);
74  // TODO - find the correct way to define large enough GATT queue for extended GATT event.
75  ble_cfg_t conn_cfg = { 0 };
76  conn_cfg.conn_cfg.conn_cfg_tag = RUUVI_NRF5_SDK15_BLE4_STACK_CONN_TAG;
77  err_code |= sd_ble_cfg_set (BLE_CONN_CFG_GATTS, &conn_cfg, ram_start);
78  // Enable BLE stack.
79  err_code |= nrf_sdh_ble_enable (&ram_start);
80  // Enable connection event extension for faster data rate
81  static ble_opt_t opt = {0};
82  opt.common_opt.conn_evt_ext.enable = true;
83  err_code |= sd_ble_opt_set (BLE_COMMON_OPT_CONN_EVT_EXT, &opt);
84  RD_ERROR_CHECK (err_code, NRF_SUCCESS);
85  // Initialize radio interrupts
86  err_code |= ble_radio_notification_init (RUUVI_NRF5_SDK15_RADIO_IRQ_PRIORITY,
87  NRF_RADIO_NOTIFICATION_DISTANCE_800US,
88  on_radio_evt);
89  // Store desired modulation
90  m_modulation = modulation;
91  }
92 
93  return ruuvi_nrf5_sdk15_to_ruuvi_error (err_code) | status;
94 }
