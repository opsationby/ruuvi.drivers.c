363 static rd_status_t set_phy_type (const ri_comm_message_t * const p_message,
364  advertisement_t * const p_adv)
365 {
366  rd_status_t err_code = RD_SUCCESS;
367  ri_radio_modulation_t modulation;
368  bool extended_required = false;
369  bool sec_phy_required = false;
370 
371  if ( (NULL == p_message) || (NULL == p_adv))
372  {
373  err_code |= RD_ERROR_NULL;
374  }
375  else
376  {
377  err_code |= ri_radio_get_modulation (&modulation);
378 
379  if (p_message->data_length > NONEXTENDED_ADV_MAX_LEN
380  || RI_RADIO_BLE_2MBPS == modulation)
381  {
382  sec_phy_required = true;
383  extended_required = true;
384  }
385 
386  switch (modulation)
387  {
388  case RI_RADIO_BLE_125KBPS:
389  p_adv->params.primary_phy = BLE_GAP_PHY_CODED;
390 
391  if (sec_phy_required)
392  {
393  p_adv->params.secondary_phy = BLE_GAP_PHY_CODED;
394  }
395 
396  extended_required = true;
397  break;
398 
399  case RI_RADIO_BLE_1MBPS:
400  p_adv->params.primary_phy = BLE_GAP_PHY_1MBPS;
401 
402  if (sec_phy_required)
403  {
404  p_adv->params.secondary_phy = BLE_GAP_PHY_1MBPS;
405  }
406 
407  break;
408 
409  case RI_RADIO_BLE_2MBPS:
410  p_adv->params.primary_phy = BLE_GAP_PHY_AUTO;
411 
412  if (sec_phy_required)
413  {
414  p_adv->params.secondary_phy = BLE_GAP_PHY_2MBPS;
415  }
416 
417  extended_required = true;
418  break;
419  }
420 
421  switch (m_type)
422  {
423  case NONCONNECTABLE_NONSCANNABLE:
424  if (extended_required)
425  {
426  p_adv->params.properties.type =
427  BLE_GAP_ADV_TYPE_EXTENDED_NONCONNECTABLE_NONSCANNABLE_UNDIRECTED;
428  }
429  else
430  {
431  p_adv->params.properties.type = BLE_GAP_ADV_TYPE_NONCONNECTABLE_NONSCANNABLE_UNDIRECTED;
432  }
433 
434  break;
435 
436  case NONCONNECTABLE_SCANNABLE:
437  if (extended_required)
438  {
439  p_adv->params.properties.type =
440  BLE_GAP_ADV_TYPE_EXTENDED_NONCONNECTABLE_SCANNABLE_UNDIRECTED;
441  }
442  else
443  {
444  p_adv->params.properties.type = BLE_GAP_ADV_TYPE_NONCONNECTABLE_SCANNABLE_UNDIRECTED;
445  }
446 
447  break;
448 
449  case CONNECTABLE_SCANNABLE:
450  if (extended_required)
451  {
452  // Cannot put extended payload and scan response into secondary PHY at the same time.
453  err_code |= RD_ERROR_DATA_SIZE;
454  LOGE ("Error: Too long data for a scannable packet.\r\n");
455  }
456  else
457  {
458  p_adv->params.properties.type = BLE_GAP_ADV_TYPE_CONNECTABLE_SCANNABLE_UNDIRECTED;
459  }
460 
461  break;
462 
463  case CONNECTABLE_NONSCANNABLE:
464  if (extended_required)
465  {
466  p_adv->params.properties.type =
467  BLE_GAP_ADV_TYPE_EXTENDED_CONNECTABLE_NONSCANNABLE_UNDIRECTED;
468  }
469  else
470  {
471  // Cannot have non-scannable connectable event on primary PHY, fallback on extended advertisement.
472  LOGW ("Warning: Fallback to BLE 5.0 done.\r\n");
473  p_adv->params.properties.type =
474  BLE_GAP_ADV_TYPE_EXTENDED_CONNECTABLE_NONSCANNABLE_UNDIRECTED;
475  }
476 
477  break;
478  }
479  }
480 
481  return err_code;
482 }

495 static rd_status_t ri_adv_send (ri_comm_message_t * message)
496 {

499
