335 rd_status_t ri_flash_record_set (const uint32_t page_id,
336  const uint32_t record_id, const size_t data_size, const void * const data)
337 {
338  if (NULL == data) { return RD_ERROR_NULL; }
339 
340  if (false == m_fds_initialized) { return RD_ERROR_INVALID_STATE; }
341 
342  /* Wait for process to complete */
343  if (m_fds_processing) { return RD_ERROR_BUSY; }
344 
345  rd_status_t err_code = RD_SUCCESS;
346  fds_record_desc_t desc = {0};
347  fds_find_token_t tok = {0};
348  /* A record structure. */
349  fds_record_t const record =
350  {
351  .file_id = page_id,
352  .key = record_id,
353  .data.p_data = data,
354  /* The length of a record is always expressed in 4-byte units (words). */
355  .data.length_words = (data_size + 3) / sizeof (uint32_t),
356  };
357  ret_code_t rc = fds_record_find (page_id, record_id, &desc, &tok);
358 
359  // If record was found
360  if (FDS_SUCCESS == rc)
361  {
362  /* Start write */
363  m_fds_processing = true;
364  rc = fds_record_update (&desc, &record);
365  err_code |= fds_to_ruuvi_error (rc);
366 
367  if (RD_SUCCESS != err_code)
368  {
369  m_fds_processing = false;
370  return err_code;
371  }
372  }
373  // If record was not found
374  else
375  {
376  /* Start write */
377  m_fds_processing = true;
378  desc.record_id = record_id;
379  rc = fds_record_write (&desc, &record);
380  err_code |= fds_to_ruuvi_error (rc);
381 
382  if (RD_SUCCESS != err_code)
383  {
384  m_fds_processing = false;
385  return err_code;
386  }
387  }
388 
389  return err_code;
390 }

405 rd_status_t ri_flash_record_get (const uint32_t page_id,
406  const uint32_t record_id, const size_t data_size, void * const data)
407 {
408  if (NULL == data) { return RD_ERROR_NULL; }
409 
410  if (false == m_fds_initialized) { return RD_ERROR_INVALID_STATE; }
411 
412  if (m_fds_processing) { return RD_ERROR_BUSY; }
413 
414  rd_status_t err_code = RD_SUCCESS;
415  fds_record_desc_t desc = {0};
416  fds_find_token_t tok = {0};
417  ret_code_t rc = fds_record_find (page_id, record_id, &desc, &tok);
418  err_code |= fds_to_ruuvi_error (rc);
419 
420  // If file was found
421  if (FDS_SUCCESS == rc)
422  {
423  fds_flash_record_t record = {0};
424  /* Open the record and read its contents. */
425  rc = fds_record_open (&desc, &record);
426  err_code |= fds_to_ruuvi_error (rc);
427 
428  // Check length
429  if (record.p_header->length_words * 4 > data_size) { return RD_ERROR_DATA_SIZE; }
430 
431  /* Copy the data from flash into RAM. */
432  memcpy (data, record.p_data, record.p_header->length_words * 4);
433  /* Close the record when done reading. */
434  rc = fds_record_close (&desc);
435  err_code |= fds_to_ruuvi_error (rc);
436  }
437 
438  return err_code;
439 }
